system{{{
	native{{{
		reg query "HKLM\Software\Microsoft\Windows NT\CurrentVersion" ## display system information
		systeminfo ## display system information		
		wmic os get osarchitechture	## shows which acrhitecture (32 or 64-bit)
		wmic computersystem get totalphysicalmemory	## display total memory installed
		reg query "hklm\system\currentcontrolset\control\timezoneinformation" ## display timezone info
		w32tm /tz ## display timezone info
		wmic os get servicepackmajorversion	## display os version number
		bcdedit /enum ## display boot config info
		reg query hklm\system\currentcontrolset\services\tcpip\parameters ## displays network settings such as the hostname, domain name, DNS servers, DHCP configuration, and other IP stack settings stored under the TCPIP parameters key
			## look for "NV Domain Key" and compair with systeminfo:
			## doesn't exist or exists and is empty = definitely workgroup
			## matches systeminfo = definitely domain
			## does *not* match systeminfo = definitely workgroup
		manage-bde -status	## view configuration for bitlocker
		
	}}}	
	sysinternals{{{
		psinfo -accepteula -d ## Show disk volume information: drive letter, format, capacity.
		psinfo -accepteula -s ## Show list of installed applications.
		psinfo -accepteula -h ## Show list of installed hotfixes.
	}}}
	powershell{{{
		powershell -c "get-computerinfo" ## display system information
		powershell -c "$env:processor_architecture"	## shows which acrhitecture (32 or 64-bit)
		powershell -c "(systeminfo | Select-String 'Total Physical Memory:').ToString().Split(':')[1].Trim()" ## display total memory installed
		powershell -c "get-timezone" ## display timezone info
		powershell -c "[environment]::OSVersion.Version" ## display os version info
		powershell -c "Get-ComputerInfo | select-object -Property csdomain,csdomainrole,csworkgroup,csroles | format-list" ## display workgroup/domain membership and roles
	}}}

users/groups{{{
	native{{{
		date /t && time /t ## display the current system date and time
		net user (<name>) ## display local users | specified user information 
		net localgroup (<groupname>) ##display local groups | specified group membership
		query user ##show logged on users and session info
		reg query hklm\system\currentcontrolset\control\hivelist ## displays a list of all currently loaded registry hives including any user-specific hives, displaying name and file path for each.
		wmic useraccount get name,groups,sid ## List users with their SIDs and associated groups
		wmic useraccount get name,fullname,sid ## List user, the full name of the user, and their SID
		whoami /groups ## list the associated groups for the user logged in as
		wmic path win32_UserProfile get localpath,sid ## displays user home directory paths and the users SID
		WMIC PATH win32_UserProfile WHERE LocalPath="c:\\users\\%username%" GET Status	## dislay current user profile status code:
			## 	0: Undefined -The status of the profile is not set.
			## 	1: Temporary-The profile is a temporary profile and will be deleted after the user logs off.
			## 	2: Roaming-The profile is set to roaming. If this bit is not set, the profile is set to local.
			## 	4: Mandatory-The profile is a mandatory profile.
			## 	8: Corrupted-The profile is corrupted and is not in use. The user or administrator must fix the corruption to use the profile.
		dsquery user domainroot  | dsget user -samid -display	## display domain user display names and SAMID
		dsquery user -name * -limit 0 | dsget user -samid -fn -ln	## display domain user first/last names and SAMID
		auditpol /get /category:* /user:<username> ## user specific logging policy info
		reg query HKEY_LOCAL_MACHINE\SAM\SAM\Domains\Account ## stores information on user accounts and groups.
	}}}
	sysinternals{{{
		psloggedon -accepteula ## currently logged on users
	}}}
	powershell{{{
		get-localuser (<username>) ## display local users | specified user information
		enable-localuser -name <username> ## enable user account (disable with disable-localuser)
		
	}}}

files/permissions{{{
	native{{{
		dir /a /s "<filename>" ## recursive search for files with the specified name ## must be done from partition root (c:\...) or include full path to dir to search
		dir /a /s /b c:\*.py ## search for python scripts
		dir /q <path\file>	## show file ownership
		forfiles /p <path> /m *.* /s /c "cmd /c if @fsize==339096 echo @path @fsize" 2>NUL ## for each file matching <path>\<mask(*.*)>, return the path of each file mathing the size specified redirecting errors
		reg query hklm /s /f "<filename>" ## recursive registry search for a specified file
	}}}
	sysinternals{{{
		accesschk "<user/group>" <path\filename> -accepteula ## /s for recursive check ##display permissions for specified user to specified file
		streams -accepteula <path\file> ## check file for alternate datastreams
	}}}
	powershell{{{
		powershell -c "get-item -path '<path\file>' -stream *" ## check file for alternate datastreams
		$(Get-PSDrive -PSProvider FileSystem).Root | Foreach { Get-ChildItem $_ -recurse | Get-Item -Stream * } ## search the filesystem for any files with an alternate data stream
	}}}

processes/services{{{
	native{{{
		schtasks | findstr "<Running|Ready|Disabled>" ## show the scheduled tasks with the given status
		schtasks /v | more ## show all details for every scheduled task
		tasklist /v | findstr /i /c:"network service" ## display list of all running processes running under the Network Service account
		tasklist /svc ## display currently running processes and associated services.
		tasklist /m ((<dllname.dll>) | findstr "<process>") ## display running processes/associated dlls | display processes that have the specifiled dll
		tasklist /fi "SERVICES eq <service>" ## only display proceses that have the specified service
		tasklist /fi "USERNAME eq <username|SYSTEM|NETWORK|etc> ## only display the processes running under the useraccount/service account specified
		tasklist /fi "IMAGENAME eq <name>" ## only show the specified process for the user specified
			## can be given other flags(/svc)
		tasklist /fi "SESSION eq <session number> ## only display processes running under given session number
		tasklist /v /fi "MEMUSAGE gt <40000>" ## only show processes using over amount of memory specified
		wmic process get name,executablepath,commandline /format:list | more ## display list of running processes showing the process name, commandline and the location of the executable
		wmic startup get caption,command,description,name,location /format:list | more ## display list of startup programs showing shortname, commandline, programs description, name of the program, and the location of the executable  
		dir C:\*<rvlkl>* /s /b ## recursive search for file name specified, returning full file path.
		dir /a /tc <C:\Users\Andrew\.frostwire5\cmd.dll> ## return creation time for the path\file specified.
		driverquery /si | findstr /i <true|false>	## check specified driver for signature; displays linkdate 
			## get timezone info and check for daylight savings time
		driverquery | findstr /i <>	## return module name, display name, driver type, and link date of driver specified
		reg query hklm /k /f "<task>" /s	## searches registry for task name specified, returns matching keys
			reg query <key from prior command>	## copy "ID" and do a recursive search
			reg query hklm /k /f "<{GUID}>" /s	## reg query each result to find the scheduled task.
	}}}
	sysinternals{{{
		pslist -t -accepteula ## process tree
		listdlls -accepteula | find /i "command" ## display the commandline for each running process
		listdlls -accepteula -u ## display unsigned dlls
		autorunsc -a -m -accepteula	## timestamps 
		autorunsc -l -m * -accepteula	## logon
		autorunsc -b -m -accepteula	## boot
		autorunsc -s -m -accepteula	## services
		autorunsc -t -m -accepteula	## tasksched
	}}}
	powershell{{{
		Get-ChildItem <C:\Users\Andrew\.frostwire5\cmd.dll> | select CreationTimeUTC ## returns the creation time of the file specified
		powershell -c "get-scheduledtask -taskname '<taskname>'" ## returns information for the task specified

Logs{{{
	native{{{
		security{{{
			wmic ntevent where (logfile='security' and eventidentifier='4625') get timewritten,username ## query security logs for failed logon attempts, displaying event time and associated user
			wevtutil qe security /rd:true /q:"*[System/EventID=4625]" ## query security logs for failed logon attempts and displaying the most recent first
		}}}
		application{{{
			wevtutil qe Application /q:*[System[(EventID=11724)]] | findstr /c:"<program>" ## query the application logs for successful program uninstalls and filtering for a specific program
			wmic ntevent where (Logfile='Application' and EventIdentifier='11724') get timewritten, user, message /value ## query the application logs for successful program uninstalls and displays the time of the event, associated user, and the event message in easy to read format
		}}}
	}}}
	Powershell{{{
		security{{{
			powershell.exe "Get-WinEvent -FilterHashtable @{ LogName='Security'; Id='4625'; data='<username>'}" ## query security logs for failed logon attempts for a specific user
		}}}
		application{{{
			powershell.exe "Get-WinEvent -FilterHashTable @{LogName='Application';Id=11724} | Where-Object {$_.Message -match '<program>'} | Select TimeCreated,Message | Format-List" ## query for uninstall events for a specific program and display the event time and message
		}}}
	}}}

	startup locations{{{ 
		## startup folder location for individual user accounts as well as a system-wide startup folder that will be checked regardless of which user account logs in.
			C:\Users\[Username]\AppData\Roaming\Microsoft\Windows\Start Menu\Programs\Startup ## current user
			C:\ProgramData\Microsoft\Windows\Start Menu\Programs\StartUp ## all users
		##  created by default on Windows systems:
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Run
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\RunOnce
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\Run
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\RunOnce
		## other default hives:
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\RunOnceEx ## not created by default on vista+
				## persistance can be created with:
					reg add HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\RunOnceEx\0001\Depend /v 1 /d "C:\temp\evil[.]dll"
		## used to set startup folder items for persistence:
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Explorer\User Shell Folders
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Explorer\Shell Folders
			HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\CurrentVersion\Explorer\Shell Folders
			HKEY_LOCAL_MACHINE\SOFTWARE\Microsoft\Windows\CurrentVersion\Explorer\User Shell Folders
		## automatic startup of services during boot:
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\RunServicesOnce
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\RunServicesOnce
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\RunServices
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\RunServices
		## Using policy settings to specify startup programs creates corresponding values in either of two Registry keys:
			HKEY_LOCAL_MACHINE\Software\Microsoft\Windows\CurrentVersion\Policies\Explorer\Run
			HKEY_CURRENT_USER\Software\Microsoft\Windows\CurrentVersion\Policies\Explorer\Run
