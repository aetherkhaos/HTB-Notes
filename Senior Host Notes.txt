Volatility {{{
	Memory Image info
		Volatility imageinfo -f file.dmp
	Processes
		Volatility --profile=Win7SP1x86_23418 pstree -f file.dmp ##Provides a process tree view (limited to non-hidden processes).
		Volatility --profile=Win7SP1x86_23418 pslist -f file.dmp ##Lists active processes similar to Task Manager.
		Volatility --profile =Win7SP1x86_23418 psscan -f file.dmp ##Scans for hidden processes (potentially unlinked from the standard list).
	Handles
		Volatility --profile=Win7SP1x86_23418 -f file.dmp handles --pid <pid> ##Investigates which resources a process has open handles to (specify process ID with -- pid ).
	Services
		Volatility --profile=Win7SP1x86_23418 -f file.dmp SvcScan ##Lists services with details like names and loaded modules.
		Volatility --profile=Win7SP1x86_23418 -f file.dmp GetSIDs ##Retrieves service SIDs for further investigation.
	Network:
		Volatility --profile=Win7SP1x86_23418 -f file.dmp NetScan ##Scans for network connections, revealing source/destination IPs, protocols, and open ports.
	Mutexes: 
		Volatility --profile=Win7SP1x86_23418 -f file.dmp MutantScan ##Scans for mutexes to understand process interactions and potential conflicts.
	Dump proc:
		Volatility -f file.dmp --profile=Win7SP1x86 memdump -p 2168 -D conhost/ ##The memory dump of a process will extract everything out of the current status of the process.
	Command Line:
		Volatility --profile=PROFILE cmdline -f file.dmp ##Display process command-line arguments
}}}
